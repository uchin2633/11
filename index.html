        sysMsg.innerText = msg;

        sysMsg.style.color = color;

        setTimeout(() => { sysMsg.innerText = "左右移動尋找目標！"; sysMsg.style.color = "#e65100"; }, 2000);

    }


    function startMoving(dir) {

        moveDirection = dir;

        document.getElementById("girl").classList.add('walking');

    }


    function stopMoving() {

        moveDirection = 0;

        document.getElementById("girl").classList.remove('walking');

    }


    function gameLoop() {

        if (moveDirection !== 0) {

            cameraX += moveDirection * 10;

            if (cameraX < 0) cameraX = 0;

            if (cameraX > worldWidth - viewWidth) cameraX = worldWidth - viewWidth;

            document.getElementById("world").style.transform = `translateX(${-cameraX}px)`;

        }

        animationFrame = requestAnimationFrame(gameLoop);

    }


    function spawnBoy() {

        if (document.querySelectorAll('.boy').length > 5) return;

        

        let boyX = Math.random() * (worldWidth - 200) + 100;

        let q = questions[Math.floor(Math.random() * questions.length)];

        

        if (boyBag.length === 0) {

            boyBag = [1, 2, 3, 4, 5, 6, 7];

            boyBag.sort(() => Math.random() - 0.5); 

        }

        let boyId = boyBag.pop(); 

        

        let boyDiv = document.createElement("div");

        boyDiv.className = "boy";

        boyDiv.style.left = boyX + "px";

        boyDiv.style.backgroundImage = `url('boy${boyId}.png')`;

        

        boyDiv.innerHTML = `

            <div class="status-face">😍</div>

            <div class="bubble">${q.text}</div>

        `;

        

        boyDiv.onclick = function() { shootLaser(boyDiv, q.isTrue); };

        document.getElementById("boys-container").appendChild(boyDiv);

        

        setTimeout(() => { if(boyDiv.parentElement) boyDiv.remove(); }, 12000);

    }


    function shootLaser(boyElement, isTrue) {

        let container = document.getElementById("game-container");

        let girl = document.getElementById("girl");

        let containerRect = container.getBoundingClientRect();

        let girlRect = girl.getBoundingClientRect();

        

        let startX = girlRect.left - containerRect.left + girlRect.width / 2;

        let startY = girlRect.top - containerRect.top + (girlRect.height * 0.2); 


        let boyRect = boyElement.getBoundingClientRect();

        let endX = boyRect.left - containerRect.left + boyRect.width / 2;

        let endY = boyRect.top - containerRect.top + (boyRect.height * 0.2); 


        let dx = endX - startX; let dy = endY - startY;

        let length = Math.sqrt(dx*dx + dy*dy);

        let angle = Math.atan2(dy, dx) * 180 / Math.PI;


        let laser = document.createElement("div");

        laser.className = "laser";

        laser.style.width = length + "px"; laser.style.left = startX + "px"; laser.style.top = startY + "px";

        laser.style.transform = `rotate(${angle}deg)`;

        container.appendChild(laser);


        boyElement.querySelector('.bubble').style.display = 'none';

        let statusFace = boyElement.querySelector('.status-face');

        statusFace.style.display = 'block';


        if (isTrue) {

            statusFace.innerText = "😍";

            score += 10;

            showMsg("⚡ 電眼成功！獲得校園大美女分數", "#d81b60");

        } else {

            statusFace.innerText = "💀";

            lives -= 1;

            showMsg("❌ 糟糕！電到迷思，扣生命值", "#424242");

        }


        updateUI();


        setTimeout(() => {

            laser.remove();

            boyElement.style.opacity = "0"; 

            setTimeout(() => boyElement.remove(), 400);

            checkWinLose();

        }, 1000);

    }


    function checkWinLose() {

        if (lives <= 0) { 

            endGame("💔 失去所有愛心，遊戲結束！"); 

        } else if (score >= targetScores[currentLevel - 1]) {

            if (currentLevel === 3) { 

                endGame("🎉 恭喜通關！👑<br>你贏得了【校園大美女】頭銜！"); 

            } else {

                levelClearTransition(); 

            }

        }

    }


    function levelClearTransition() {

        cancelAnimationFrame(animationFrame); 

        clearInterval(spawnTimer); 

        stopMoving(); 

        

        let overlay = document.getElementById("overlay");

        overlay.style.display = "flex";

        

        document.getElementById("overlay-title").innerText = `✨ 第 ${currentLevel} 關 通關！ ✨`;

        document.getElementById("overlay-desc").innerHTML = `太棒了！準備好迎接下一關了嗎？<br>下一關過關目標：${targetScores[currentLevel]} 分`;

        

        let btn = document.getElementById("start-btn");

        btn.innerText = "進入下一關 ➡️";

        btn.onclick = nextLevel; 

    }


    function nextLevel() {

        document.getElementById("overlay").style.display = "none";

        

        currentLevel++; 

        score = 0; 

        cameraX = 0;

        boyBag = [];

        document.getElementById("boys-container").innerHTML = ""; 

        

        document.getElementById("overlay-title").innerText = "✨ 衛教電眼美女 ✨";

        

        updateUI();

        gameLoop();

        spawnTimer = setInterval(spawnBoy, 2500);

    }


    function endGame(msg) {

        cancelAnimationFrame(animationFrame); 

        clearInterval(spawnTimer);

        stopMoving();

        

        bgMusic.pause();

        bgMusic.currentTime = 0;


        let overlay = document.getElementById("overlay");

        overlay.style.display = "flex";

        

        document.getElementById("overlay-title").innerText = "✨ 衛教電眼美女 ✨";

        document.getElementById("overlay-desc").innerHTML = msg;

        

        let btn = document.getElementById("start-btn");

        btn.innerText = "重新挑戰";

        btn.onclick = startGame; 

    }
